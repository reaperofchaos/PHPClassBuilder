80 std::vector<std::string> checkIfLineIsTooLong(std::string line)
81 {
82 std::vector<std::string> lines;
83 int i = 0;
84 int current = 0;
85 int limit = 70;
86 if(line.size() > limit)
87 {
88 while(line.substr(i, -1).size() > limit)
89 {
90 current = i;
91 i = i + limit;
92 while(line.at(i) != ' ')
93 {
94 i = i - 1;
95 }
96
97 //std::cout << "i is " << i << "/n";
98 lines.push_back(line.substr(current, i));
99 }
100 //skip the space by starting a character late
101 lines.push_back(line.substr(i+1, -1));
102 }
103 else
104 {
105 lines.push_back(line.substr(i, -1));
106 }
107 return lines;
